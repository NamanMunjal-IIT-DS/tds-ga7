import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

# Email for verification: 24f2001419@ds.study.iitm.ac.in

def load_and_analyze_employee_data():
    """
    Function to load employee data, calculate HR frequency, and create visualization
    """
    # Sample dataset creation (in real scenario, you would load from CSV)
    np.random.seed(42)  # For reproducibility

    # Define the departments and regions
    departments = ['Operations', 'Sales', 'R&D', 'HR', 'Finance', 'Marketing']
    regions = ['Africa', 'Asia Pacific', 'North America', 'Europe']

    # Generate 100 employees
    employee_data = []
    for i in range(1, 101):
        emp_id = f"EMP{i:03d}"
        department = np.random.choice(departments)
        region = np.random.choice(regions)

        # Generate performance score (50100)
        performance_score = round(np.random.uniform(50, 100), 2)

        # Generate years of experience (120)
        years_experience = np.random.randint(1, 21)

        # Generate satisfaction rating (15)
        satisfaction_rating = round(np.random.uniform(1, 5), 1)

        employee_data.append([
            emp_id,
            department,
            region,
            performance_score,
            years_experience,
            satisfaction_rating
        ])

    # Create DataFrame
    df = pd.DataFrame(
        employee_data,
        columns=[
            'employee_id',
            'department',
            'region',
            'performance_score',
            'years_experience',
            'satisfaction_rating'
        ]
    )

    # Print basic info
    print("Employee Performance Dataset Analysis")
    print("=" * 50)
    print(f"Total employees: {len(df)}\n")
    print("First 5 rows:")
    print(df.head(), "\n")

    # Calculate frequency count for HR department
    hr_frequency = df[df['department'] == 'HR'].shape[0]
    print(f"Frequency count for HR department: {hr_frequency}\n")

    # Department distribution
    department_counts = df['department'].value_counts()
    print("Department distribution:")
    print(department_counts, "\n")

    # Create histogram showing distribution of departments
    plt.figure(figsize=(12, 8))

    bars = plt.bar(
        department_counts.index,
        department_counts.values,
        color=['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FFEAA7', '#DDA0DD'],
        edgecolor='black',
        linewidth=1.2
    )
    plt.title('Distribution of Employees Across Departments',
              fontsize=16, fontweight='bold', pad=20)
    plt.xlabel('Department', fontsize=12, fontweight='bold')
    plt.ylabel('Number of Employees', fontsize=12, fontweight='bold')
    plt.xticks(rotation=45, ha='right')
    plt.grid(axis='y', alpha=0.3)

    # Add value labels
    for bar in bars:
        height = bar.get_height()
        plt.text(
            bar.get_x() + bar.get_width() / 2.,
            height + 0.5,
            f'{int(height)}',
            ha='center',
            va='bottom',
            fontweight='bold'
        )

    plt.tight_layout()
    plt.show()

    return df, hr_frequency

if __name__ == "__main__":
